# AIR WebComponents

> Standard Webcomponents for AIR projects

## Installation

```sh
yarn add @rws-air/webcomponents
```

## Usage

```tsx
import React, { FC } from 'react'

import { Button } from '@rws-air/webcomponents'

const Example: FC = () => (
  <Button label='button' onClick={() => console.log('void')} />
)
```
